Executive Summary
Entwickle eine professionelle Web-Applikation zur Generierung rechtssicherer Wohnraummietverträge nach deutschem Recht (BGB, insb. §§ 535-580a). Die App führt Vermieter durch einen mehrstufigen Wizard und generiert ein layoutgetreues PDF basierend auf dem Haus & Grund Bremen Standard.

2. Rechtslogik & Validierung
2.1 Automatische Checks (mit Warn/Error-System)
typescriptinterface ValidationRule {
  regel: string;
  typ: 'ERROR' | 'WARNING' | 'INFO';
  pruefen: (data: MietvertragDaten) => boolean;
  meldung: string;
  rechtsgrundlage?: string;
}

const VALIDATION_RULES: ValidationRule[] = [
  // ERRORS (blockieren Generierung)
  {
    regel: 'Kaution max. 3 Monatsmieten',
    typ: 'ERROR',
    pruefen: (d) => d.kaution.betrag <= d.grundmiete.summe * 3,
    meldung: 'Kaution darf max. 3 Monats-Grundmieten betragen',
    rechtsgrundlage: '§ 551 Abs. 1 BGB'
  },
  
  {
    regel: 'Befristungsgrund bei Zeitmietvertrag',
    typ: 'ERROR',
    pruefen: (d) => {
      if (d.vertragsart === 'befristet') {
        return d.befristung?.grundBeschreibung.length > 20;
      }
      return true;
    },
    meldung: 'Befristungsgrund muss gem. § 575 BGB schriftlich mitgeteilt werden',
    rechtsgrundlage: '§ 575 Abs. 1 S. 2 BGB'
  },
  
  // WARNINGS (erlaubt Generierung, warnt aber)
  {
    regel: 'Kleinreparaturklausel Einzelfall',
    typ: 'WARNING',
    pruefen: (d) => {
      if (!d.kleinreparaturen.aktiv) return true;
      return d.kleinreparaturen.einzelfall_max <= 120;
    },
    meldung: 'BGH: Einzelfall-Grenze über 100-120€ kann unwirksam sein',
    rechtsgrundlage: 'BGH VIII ZR 78/06'
  },
  
  {
    regel: 'Kleinreparaturklausel Jahresgrenze',
    typ: 'WARNING',
    pruefen: (d) => {
      if (!d.kleinreparaturen.aktiv) return true;
      const jahresProzent = (d.kleinreparaturen.jahresmax / (d.grundmiete.summe * 12)) * 100;
      return jahresProzent <= 8;
    },
    meldung: 'BGH: Jahresgrenze sollte ca. 6-8% der Jahreskaltmiete nicht überschreiten',
    rechtsgrundlage: 'BGH VIII ZR 242/06'
  },
  
  {
    regel: 'Schönheitsreparaturen nur bei übergabe renoviert',
    typ: 'WARNING',
    pruefen: (d) => {
      if (!d.schoenheitsreparaturen.uebertragen) return true;
      return d.schoenheitsreparaturen.zustand_uebergabe === 'renoviert';
    },
    meldung: 'Übertragung der Schönheitsreparaturen kann unwirksam sein bei unrenovierter Übergabe',
    rechtsgrundlage: 'BGH VIII ZR 185/14'
  },
  
  // INFO
  {
    regel: 'Staffelmiete: Min. 12 Monate Abstand',
    typ: 'INFO',
    pruefen: (d) => {
      if (d.mieterhoehung.typ !== 'staffel') return true;
      // Prüfe Abstände
      const dates = d.mieterhoehung.staffel.erhoehungen.map(e => e.datum);
      for (let i = 1; i < dates.length; i++) {
        const diff = (dates[i].getTime() - dates[i-1].getTime()) / (1000 * 60 * 60 * 24);
        if (diff < 365) return false;
      }
      return true;
    },
    meldung: 'Staffelmiete muss mind. 12 Monate unverändert bleiben',
    rechtsgrundlage: '§ 557a Abs. 2 BGB'
  }
];

typescript// api/generate-pdf.ts
import puppeteer from 'puppeteer';
import { renderToStaticMarkup } from 'react-dom/server';

export async function generateMietvertragPDF(data: MietvertragDaten) {
  const html = renderToStaticMarkup(<MietvertragHTML data={data} />);
  
  const browser = await puppeteer.launch();
  const page = await browser.newPage();
  
  await page.setContent(html, { waitUntil: 'networkidle0' });
  
  const pdf = await page.pdf({
    format: 'A4',
    printBackground: true,
    margin: {
      top: '20mm',
      bottom: '20mm',
      left: '15mm',
      right: '15mm'
    },
    displayHeaderFooter: true,
    headerTemplate: `<div style="font-size:10px; width:100%; text-align:center;">
      Mietvertrag für Wohnraum
    </div>`,
    footerTemplate: `<div style="font-size:9px; width:100%; text-align:center;">
      Seite <span class="pageNumber"></span> von <span class="totalPages"></span>
    </div>`
  });
  
  await browser.close();
  return pdf;
}
